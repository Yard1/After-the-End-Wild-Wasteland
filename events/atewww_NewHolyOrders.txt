# The daughter of a ruler volunteers to join the Order of Lady Liberty
character_event = {
	id = NewHolyOrders.1
	hide_window = yes
	
	only_women = yes
	min_age = 16
	max_age = 45
	only_capable = yes
	prisoner = no
	religion_group = oldworld
	
	trigger = {
		is_ruler = no
		d_libertarian = {
			holy_order_can_PREV_join = yes
		}
		
		martial = 5
				
		OR = {
			father_even_if_dead = {
				primary_title = { higher_tier_than = BARON }
			}
			mother_even_if_dead = {
				primary_title = { higher_tier_than = BARON }
			}
		}
		
		liege = {
			holy_order = no
            OR = {
                religion = americanist
                religion = hamiltonian
                religion = jeffersonian
                religion = libertarian
            }
			liege = {
				holy_order = no
			}
		}
		
		
		
		NOT = { has_character_modifier = expelled_d_libertarian }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_libertarian
			}
		}

		OR = {
			AND = {
				is_female = no
				NOT = {
					any_spouse = {
						OR = {
							is_landed = yes
							AND = {
								is_pregnant = yes
								father_of_unborn = { character = ROOT } # Won't happen if spouse is pregnant with non-bastard child
							}
						}
					}
				}
			}
			AND = {
				is_female = yes
				NOR = {
					is_landed = yes
					is_pregnant = yes #Simply won't join if pregnant
				}
			}
		}
	}
	
	has_dlc = "Sons of Abraham"
	mean_time_to_happen = {
		years = 80
		modifier = {
			factor = 0.75
			liege = {
				society_member_of = monastic_order_libertarian
			}
		}
		modifier = {
			factor = 0.25
			society_member_of = monastic_order_libertarian
		}
		modifier = {
			factor = 0.25
			trait = zealous
		}
		modifier = {
			factor = 0.5
			trait = celibate
		}
		modifier = {
			factor = 0.5
			trait = homosexual
		}
		modifier = {
			factor = 0.7
			trait = content
		}
		modifier = {
			factor = 0.75
			trait = chaste
		}
		modifier = {
			factor = 0.75
			NOT = { age = 25 }
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 2.0
			trait = lustful
		}
		modifier = {
			factor = 2.0
			trait = cynical
		}
		modifier = {
			factor = 2.0
			trait = hedonist
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			any_heir_title = {
				always = yes
			}
		}
		modifier = {
			factor = 2.0
			is_primary_heir = yes
		}
		modifier = {
			factor = 2
			any_spouse = {
				reverse_opinion = { who = ROOT value = 100 }
			}
		}
		modifier = {
			factor = 2
			d_libertarian = {
				holder_scope = { NOT = { is_within_diplo_range = ROOT } }
			}
		}
	}
	
	immediate = {
		liege = {
			character_event = {
				id = NewHolyOrders.2
			}
		}
	}
}

# Son or brother asks to join the Order of Lady Liberty
character_event = {
	id = NewHolyOrders.2
	desc = EVTDESC_ATEWWNHO_2
	picture = GFX_evt_crusaders
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_SOA_1031 # Agree
		ai_chance = {
			factor = 100
		}
		
		if = {
			limit = {
				religion = FROM
			}
			piety = 100
		}
		if = {
			limit = {
				society_member_of = monastic_order_libertarian
			}
			add_society_currency_major_effect = yes
		}
		
		FROM = {
			character_event = {
				id = NewHolyOrders.3
				tooltip = EVTTOOLTIP_ATEWWNHO_2
			}
		}
	}
	
	option = {
		name = EVTOPTB_SOA_1031 # Refuse
		
		if = {
			limit = {
				religion = FROM
			}
			piety = -100
		}
		if = {
			limit = {
				society_member_of = monastic_order_libertarian
			}
			detract_society_currency_major_effect = yes
		}
		
		FROM = {
			character_event = {
				id = NewHolyOrders.4
				tooltip = EVTTOOLTIP_SOA_1033
			}
		}
	}
}

# The Grand Master humbly requests the right to build a castle in your demesne
character_event = {
	id = NewHolyOrders.7
	hide_window = yes
	
	min_age = 16
	only_rulers = yes
	religion_group = oldworld
	
	trigger = {
		NAND = {
			ai = no
			society_member_of = monastic_order_libertarian # they get quests
		}
        d_libertarian = {
			holder_scope = {
				character = yes
				religion = ROOT
				OR = {
					is_vassal_or_below_of = ROOT
					independent = yes
				}
			}
		}
		d_libertarian = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
		}
		
		
		NOT = { has_character_modifier = expelled_d_libertarian }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_libertarian
			}
		}
	}
	
	has_dlc = "Sons of Abraham"
	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 0.5
			d_libertarian = {
				holder_scope = {
					is_vassal_or_below_of = ROOT
				}
			}
		}
		modifier = {
			factor = 0.5
			d_libertarian = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_libertarian = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_libertarian = {
				holder_scope = {
					num_of_count_titles = 1
				}
			}
		}
		modifier = {
			factor = 2.0
			d_libertarian = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
		modifier = {
			factor = 1.5
			ai = yes
			society_member_of = monastic_order_libertarian # they get quests
		}
		modifier = {
			factor = 2
			d_libertarian = {
				holder_scope = { NOT = { is_within_diplo_range = ROOT } }
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
			}
			preferred_limit = {
				any_province_holding = {
					is_holy_site = ROOT
				}
			}
			preferred_limit = {
				religion = ROOT
			}
			province_event = {
				id = NewHolyOrders.8
			}
		}
	}
}
