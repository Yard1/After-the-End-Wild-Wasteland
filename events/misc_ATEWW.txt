namespace = miscATEWW

character_event = {
	id = miscATEWW.1
	desc = "Make sure newly spawned characters in AI holy orders/theocracies make sense"
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		FROMFROM = {
			ai = yes
			OR = {
				holy_order = yes
				is_theocracy = yes
				mercenary = yes
				primary_title = { is_landless_type_title = yes }
			}
			primary_title = { 
				NOT = { 
					has_law = agnatic_succession 
				}
			}
		}
	}
	
	option = {
		log = "miscATEWW.1: [Root.GetBestName] in [FromFrom.GetBestName] ([FromFrom.Religion.GetName])"
		if = {
			limit = {
				FROM = { character = no }
			}
			if = {
				limit = {
					FROMFROM = { 
						OR = {
							holy_order = yes
							mercenary = yes
						}
					}
				}
				log = "miscATEWW.1: [FromFrom.GetBestName] is holy order or mercenary"
				#all female
				if = {
					limit = {
						FROMFROM = {
							primary_title = { 
								OR = {
									has_law = enatic_succession
									has_law = enatic_cognatic_succession
								}
							}
						}
					}
					log = "miscATEWW.1: [Root.GetBestName] holy order/merc"
					set_gender = female
				}
				else = {
					#60/40
					log = "miscATEWW.1: [Root.GetBestName] 60/40 holy order/merc"
					character_event = { id = miscATEWW.5 }
				}
			}
			else_if = {
				limit = {
					OR = {
						is_priest = yes
						trait = detached_priest
						trait = martial_cleric
						trait = scholarly_theologian
						trait = mastermind_theologian
					}
				}
				if = {
					limit = {
						FROMFROM = {
							religion_allows_female_temple_holders = yes
							religion_allows_male_temple_holders = no
						}
					}
					log = "miscATEWW.1: [Root.GetBestName] priest"
					set_gender = female
				}
				else_if = {
					limit = {
						FROMFROM = {
							religion_allows_female_temple_holders = yes
							religion_allows_male_temple_holders = yes
						}
					}
					log = "miscATEWW.1: [Root.GetBestName] 60/40 priest"
					character_event = { id = miscATEWW.5 }
				}
			}
			else_if = {
				limit = {
					FROMFROM = { matriarchal_religion = yes }
				}
				log = "miscATEWW.1: [Root.GetBestName] 80/20"
				set_character_flag = 80_20_female
				character_event = { id = miscATEWW.5 }
			}
			else = {
				log = "miscATEWW.1: [Root.GetBestName] 60/40"
				character_event = { id = miscATEWW.5 }
			}
		}
		if = {
			limit = { is_female = yes }
			if = {
				limit = {
					religion = libertarian
				}
				give_ROOT_random_female_libertarian_name = yes
			}
			else = {
				random_character = {
					limit = { 
						NOT = { character = PREV }
						is_female = yes
						culture = ROOT
					}
					preferred_limit = {
						religion = ROOT
					}
					ROOT = { copy_name = PREV }
				}
			}
		}
		if = {
			limit = {
				religion = libertarian
			}
			character_event = { id = miscATEWW.3 days = 1 }
		}
		if = {
			limit = {
				FROMFROM = { has_landed_title = d_nancy_harts }
				NOT = { has_trait = shieldmaiden }
				martial = 5
			}
			random = {
				chance = 20
				modifier = {
					factor = 2
					martial = 10
				}
				modifier = {
					factor = 2
					martial = 20
				}
				add_trait = shieldmaiden
			}
		}
		if = {
			limit = {
				FROMFROM = { 
					OR = {
						holy_order = yes
						mercenary = yes
					}
				}
			}
			FROMFROM = {
				ROOT = { clear_wealth = PREV }
				character_event = { id = miscATEWW.2 }
			}
		}
	}
}

character_event = {
	id = miscATEWW.2
	desc = "Make sure newly spawned characters in AI holy orders/theocracies make sense"
	is_triggered_only = yes
	hide_window = yes
	only_rulers = yes
	trigger = {
		OR = {
			holy_order = yes
			mercenary = yes
		}
	}
	
	option = {
		log = "miscATEWW.2: Fire for [Root.GetTitledName] ([Root.Religion.GetName])"
		if = {
			limit = {
				holy_order = yes
				primary_title = {
					holy_order_is_celibate = yes
				}
				NOT = { has_trait = celibate }
			}
			spouse = {
				remove_spouse = spouse
				character_event = { id = SoA.2863 }
			}
			add_trait = celibate
		}
		if = {
			limit = {
				has_landed_title = d_nancy_harts
				NOT = { has_trait = shieldmaiden }
			}
			add_trait = shieldmaiden
		}
		if = {
			limit = { has_landed_title = d_men_in_black }
			primary_title = {
				if = {
					limit = {
						NOT = { has_law = succ_mib_elective }
					}
					add_law = succ_mib_elective
				}
			}
			if = {
				limit = { 
					independent = yes 
					NOR = {  
						religion = americanist
						religion = jeffersonian
						religion = hamiltonian
						religion = libertarian
					}
				}
				religion = americanist
				abandon_heresy = yes
			}
			else_if = {
				limit = {
					liege = {
						NOR = {
							religion = americanist
							religion = jeffersonian
							religion = hamiltonian
							religion = libertarian
						}
					}
					NOR = { 
						true_religion = americanist 
						true_religion = jeffersonian
						true_religion = hamiltonian
						true_religion = libertarian
					}
				}
				clear_secret_religion = yes
				set_secret_religion = americanist
			}

			#Try and force them to lvl 3 of the assassins
			if = {
				limit = { NOT = { society_member_of = the_assassins } }
				join_society = the_assassins
			}
			if = {
				limit = {
					society_rank = {
						rank == 1
						society = the_assassins
					}
				}
				society_rank_up = {
					rank = 2
					society = the_assassins
				}
			}
			else_if = {
				limit = {
					society_rank = {
						rank == 2
						society = the_assassins
					}
				}
				society_rank_up = {
					rank = 1
					society = the_assassins
				}
			}
		}
		if = {
			limit = { has_landed_title = d_libertarian }
			log = "miscATEWW.2: [Root.GetTitledName] in libertarian block"
			primary_title = {
				if = {
					limit = {
						NOT = { has_law = succ_libertarian_elective }
					}
					add_law = succ_libertarian_elective
				}
			}
			#Changes religion
			if = {
				limit = { 
					NOT  = { religion = libertarian }
				}
				religion = libertarian
			}

			#Try and force them to lvl 4 of the order
			if = {
				limit = { NOT = { society_member_of = monastic_order_libertarian } }
				join_society = monastic_order_libertarian
			}
			if = {
				limit = {
					society_rank = {
						rank == 1
						society = monastic_order_libertarian
					}
				}
				society_rank_up = {
					rank = 3
					society = monastic_order_libertarian
				}
			}
			else_if = {
				limit = {
					society_rank = {
						rank == 2
						society = monastic_order_libertarian
					}
				}
				society_rank_up = {
					rank = 2
					society = monastic_order_libertarian
				}
			}
			else_if = {
				limit = {
					society_rank = {
						rank == 3
						society = monastic_order_libertarian
					}
				}
				society_rank_up = {
					rank = 1
					society = monastic_order_libertarian
				}
			}
			if = {
				limit = { is_society_grandmaster = no }
				set_society_grandmaster = yes
			}
		}
		any_realm_character = {
			if = {
				limit = {
					holy_order = yes
					ROOT = {
						holy_order = yes
						primary_title = {
							holy_order_is_celibate = yes
						}
					}
					NOT = { has_trait = celibate }
				}
				spouse = {
					remove_spouse = spouse
					character_event = { id = SoA.2863 }
				}
				add_trait = celibate
			}
			if = {
				limit = {
					holy_order = yes
					ROOT = {
						has_landed_title = d_men_in_black
						OR = {
							true_religion = americanist
							true_religion = jeffersonian
							true_religion = hamiltonian
							true_religion = libertarian
						}
					}
				}
				if = {
					limit = {
						NOR = {
							true_religion = americanist
							true_religion = jeffersonian
							true_religion = hamiltonian
							true_religion = libertarian
						}
					}
					if = {
						limit = {
							liege = {
								NOR = {
									religion = americanist
									religion = jeffersonian
									religion = hamiltonian
									religion = libertarian
								}
							}
							OR = { 
								true_religion = americanist 
								true_religion = jeffersonian
								true_religion = hamiltonian
								true_religion = libertarian
							}
						}
						clear_secret_religion = yes
						set_secret_religion = americanist
					}
					else = {
						religion = americanist
						abandon_heresy = yes
					}
				}
				if = {
					limit = {
						is_adult = yes
						prisoner = no
						NOT = { trait = incapable }
						NOT = { society_member_of = the_assassins }
						can_join_society = the_assassins
					}
					join_society = the_assassins
				}
			}
			else_if = {
				limit = {
					holy_order = yes
					ROOT = {
						has_landed_title = d_libertarian
						religion = libertarian
					}
				}
				if = {
					limit = {
						NOT = {
							religion = libertarian
						}
					}
					religion = libertarian
				}
				if = {
					limit = {
						is_female = yes
						is_adult = yes
						prisoner = no
						NOT = { trait = incapable }
					}
					if = {
						limit = {
							NOT = { society_member_of = monastic_order_libertarian }
							can_join_society = monastic_order_libertarian
						}
						join_society = monastic_order_libertarian
						if = {
							limit = {
								society_rank = {
									rank == 1
									society = monastic_order_libertarian
								}
							}
							society_rank_up = {
								rank = 1
								society = monastic_order_libertarian
							}
						}
					}
					if = {
						limit = {
							society_rank = {
								rank == 2
								society = monastic_order_libertarian
							}
							NOT = { year = 2668 }
						}
						random_list = {
							4 = {
							}
							6 = {
								trigger = {
									society_can_rank_up = yes
								}
								modifier = {
									factor = 2
									has_minor_title = title_commander
								}
								society_rank_up = {
									rank = 1
									society = monastic_order_libertarian
								}
							}
							1 = {
								trigger = {
									society_can_rank_up = yes
								}
								modifier = {
									factor = 2
									OR = {
										has_job_title = job_chancellor
										has_job_title = job_marshal
										has_job_title = job_treasurer
										has_job_title = job_spymaster
										has_job_title = job_spiritual
									}
								}
								society_rank_up = {
									rank = 1
									society = monastic_order_libertarian
								}
								if = {
									limit = { society_can_rank_up = yes }
									society_rank_up = {
										rank = 1
										society = monastic_order_libertarian
									}
								}
							}
						}
					}
				}
			}
		}
	}
}
