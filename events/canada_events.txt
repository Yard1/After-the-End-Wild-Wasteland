		hidden_tooltip = {

					}
				}

						has_landed_title = k_ursuline

							has_character_flag = was_ursuline

			}

				job_spiritual = {
					save_event_target_as = new_pope
				}
			}
			event_target:new_pope = {
				any_vassal = {
					set_character_flag = old_new_pope_vassal

							title = k_confederated

					who = event_target:unifier

					set_character_flag = old_new_pope_vassal
				}

					limit = {
						NOR = {
							title = k_confederated
							title = k_ursuline
							title = d_high_church 
						}
					}
					set_title_flag = old_new_pope_title
				}
				character_event = { id = canada.999 }
			}
			if = {
				limit = {
					d_high_church = {
						holder_scope = {
							is_liege_or_above = ROOT
						}
					}
				}
				d_high_church = {
					holder_scope = {
						save_event_target_as = old_bishop
						any_vassal = {
							set_character_flag = old_bishop_vassal
						}
